%----

\newcommand{\AdventTitleBlock}[4]{%
  \begin{center}
    {\Large\textcolor{adventred}{\textbf{#1}}}\par\vspace{4pt}%
    \ifx&#2&\else
      {\large\textbf{#2}}\par\vspace{2pt}%
    \fi
    {\Large\textcolor{adventblue}{\textbf{#3}}}\par
    \ifx&#4&\else
      \vspace{2pt}%
      {\normalsize\textbf{#4}}\par%
    \fi
  \end{center}%
}

\newcommand{\AdventKeyInsight}[1]{%
  \vspace{0.5em}%
  \noindent\colorbox{adventred!8}{%
    \parbox{\dimexpr\linewidth-2\fboxsep}{%

    }%
  }%
  \vspace{0.5em}%
}

  \end{center}
  \vspace{0.3em}%
}

%---- AdventSheet macro (1-page, single column) ----
% #1: Date + occasion
% #2: unused
% #3: Main title
% #4: Subtitle
% #5: Key Insight
% #6: Main body content
% #7: Closing statement
\newcommand{\AdventSheet}[7]{%
  \BeginAdventPage
  \vspace*{1cm}
  \begin{AdventFrameTop}
    \AdventTitleBlock{#1}{#2}{#3}{#4}
    \AdventKeyInsight{#5}
  \end{AdventFrameTop}
  \AdventStarRule
  #6%
  \EndAdventPage
  \AdventClosing{#7}%
}

%---- AdventSheetTwoCol macro (two-column hero sheet) ----
% #1: Date + occasion
% #2: unused
% #3: Main title
% #4: Subtitle
% #5: Key Insight
% #6: Main body content (wrapped in multicols{2})
% #7: Closing statement
\newcommand{\AdventSheetTwoCol}[7]{%
  \BeginAdventPage
  \begin{AdventFrameTop}
    \AdventTitleBlock{#1}{#2}{#3}{#4}
    \AdventKeyInsight{#5}
  \end{AdventFrameTop}
  \AdventStarRule
  \begin{multicols}{2}
    #6%
  \end{multicols}
  \EndAdventPage
  \AdventClosing{#7}%
}
